<objective>
Stream git clone and git pull progress in real-time to the frontend log viewer by replacing buffered `.output().await` with `.spawn()` + line-by-line stderr reading.

Purpose: Currently `git clone --progress` outputs progress to stderr, but `.output()` buffers all output until the process exits. The user sees nothing until the clone completes. This change emits each progress line immediately via the existing `emit_log()` function.

Output: Modified `docker_install.rs` with real-time stderr streaming for both git clone and git pull paths.

2. Replace the `let output = tokio::process::Command::new("git").args([...]).output().await` pattern with:
```rust
let mut child = tokio::process::Command::new("git")
    .args([
        "clone",
        "--depth", "1",
        "--progress",
        OPENCLAW_REPO,
        repo_dir.to_str().unwrap(),
    ])
    .stdout(Stdio::piped())
    .stderr(Stdio::piped())
    .spawn()
    .map_err(|e| AppError::InstallationFailed {
        reason: format!("Failed to run git clone: {e}"),
        suggestion: "Ensure git is installed and on your PATH".into(),
    })?;

// Stream stderr lines in real-time (git --progress outputs to stderr)
let stderr_handle = child.stderr.take().unwrap();
let mut stderr_reader = BufReader::new(stderr_handle).lines();
let stderr_task = tokio::spawn(async move {
    let mut lines = Vec::new();
    while let Ok(Some(line)) = stderr_reader.next_line().await {
        lines.push(line);
    }
    lines
});

// Also drain stdout
let stdout_handle = child.stdout.take().unwrap();
let mut stdout_reader = BufReader::new(stdout_handle).lines();
while let Ok(Some(line)) = stdout_reader.next_line().await {
    emit_log(app_handle, &line);
}

// Wait for process to finish
let status = child.wait().await.map_err(|e| AppError::InstallationFailed {
    reason: format!("git clone process error: {e}"),
    suggestion: "Check your internet connection. Ensure git is installed.".into(),
})?;

// Collect stderr lines from background task
let stderr_lines = stderr_task.await.unwrap_or_default();
for line in &stderr_lines {
    emit_log(app_handle, line);
}

if !status.success() {
    let stderr = stderr_lines.join("\n");
    return Err(AppError::InstallationFailed {
        reason: format!("git clone failed: {stderr}"),
        suggestion: "Check your internet connection. Ensure git is installed.".into(),
    });
}

emit_log(app_handle, "Repository cloned successfully.");
```

Key points:
- Use `Stdio::piped()` on both stdout and stderr
- Read stderr in a `tokio::spawn` task so it streams concurrently while the process runs
- Read stdout in the main task (typically empty for git clone, but drain it)
- Collect stderr lines for error reporting if the process fails
- Add `use std::process::Stdio;` import if not present

Also apply the same pattern to the **git pull block** (lines 114-127):
